    callback_error: function (element) {
        // 保存原始图片 URL
        const originalSrc = element.getAttribute('data-src');
        
        // 设置失败图片
        element.src = '/resource/image/click_reload.jpg';
        
        // 添加点击重新加载事件
        element.addEventListener('click', function reloadImage() {
            // 设置加载中图片
            element.src = '/resource/image/loading.svg';
            
            // 创建新的图片对象来加载
            const newImage = new Image();
            newImage.src = originalSrc;
            
            newImage.onload = function() {
                // 加载成功后更新图片
                element.src = originalSrc;
                // 移除点击事件（防止重复绑定）
                element.removeEventListener('click', reloadImage);
            };
            
            newImage.onerror = function() {
                // 如果再次加载失败，恢复到点击重新加载的状态
                element.src = '/resource/image/click_reload.jpg';
            };
        });
    }
});
</script>
